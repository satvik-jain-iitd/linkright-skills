  required directories, writes user_setup.md from user answers, installs Python deps,
  verifies system tools (git, gh, sqlite3, TTS), scaffolds GitHub repos (linkright-resume,
  linkright-portfolio, linkright-memory), initializes jobs DB, creates profile embedding
  placeholder, writes .env for API keys, and runs a full health check across all 8 skills.
  Also runs as a repair tool when any skill breaks — re-checks only failed components.

  Use when user says: /linkright-setup, "setup linkright", "install linkright",
  "initialize linkright", "linkright not working", "fix linkright", "health check",
  "linkright broken", "first time setup", or any setup/repair request.

## Absolute Rules

- NEVER overwrite existing user_setup.md without showing diff + explicit confirm
- NEVER create public GitHub repos without showing user exactly what will be public
- NEVER store API keys in user_setup.md — only in ~/.linkright/.env
- ALWAYS show what each step does before running it
- ALWAYS continue on non-fatal errors — report at end, don't abort

```bash
ls ~/.linkright/user_setup.md 2>/dev/null && echo "EXISTS" || echo "MISSING"
```

## Step 3 — System Tools Check

```bash
echo "=== System Tools ==="
check() {
  if command -v "$1" >/dev/null 2>&1; then
    echo "  ✓ $1 ($(command -v $1))"
  else
    echo "  ✗ $1 — MISSING. Install: $2"
  fi
}

check git      "brew install git"
check gh       "brew install gh"
check sqlite3  "brew install sqlite3"
check python3  "brew install python3"
check node     "brew install node"

if command -v say >/dev/null 2>&1; then
  echo "  ✓ TTS: macOS say (built-in)"
  say -v Samantha "LinkRight setup complete" 2>/dev/null &
elif command -v espeak-ng >/dev/null 2>&1; then
  echo "  ✓ TTS: espeak-ng"

fi
```

## Step 4 — Python Dependencies

```bash
echo "=== Python Dependencies ==="
REQUIRED=(
  "httpx"
  "curl_cffi"
  "numpy"
  "sentence_transformers"
  "openpyxl"
  "jobspy"
  "pyttsx3"
  "pypdf"
  "weasyprint"
)

MISSING=()
for pkg in "${REQUIRED[@]}"; do
  if python3 -c "import ${pkg//-/_}" 2>/dev/null; then
    echo "  ✓ $pkg"
  else
    echo "  ✗ $pkg — missing"
    MISSING+=("$pkg")
  fi
done

if [ ${#MISSING[@]} -gt 0 ]; then
  echo ""
  echo "Missing packages: ${MISSING[*]}"
  echo "Install all? (y/n)"
  read -r REPLY
  if [[ "$REPLY" =~ ^[Yy]$ ]]; then
    pip3 install "${MISSING[@]}"
    echo "Install complete ✓"
  else
    echo "Skipped. Run manually: pip3 install ${MISSING[*]}"
  fi
else
  echo "All Python deps present ✓"
fi
```

## Step 5 — Jobs DB Initialization

```bash
echo "=== Jobs DB ==="
JOBS_DIR="$HOME/.linkright/jobs"

if [ -f "$JOBS_DIR/db/jobs.db" ]; then
  COUNT=$(sqlite3 "$JOBS_DIR/db/jobs.db" "SELECT COUNT(*) FROM jobs;" 2>/dev/null || echo "0")
  echo "  ✓ jobs.db exists ($COUNT rows)"
else
  echo "  Creating jobs.db schema..."
  sqlite3 "$JOBS_DIR/db/jobs.db" "
  CREATE TABLE IF NOT EXISTS jobs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT,
    company TEXT,
    location TEXT,
    sector TEXT,
    role_category TEXT,
    relevance_score REAL,
    is_remote INTEGER DEFAULT 0,
    is_ghost INTEGER DEFAULT 0,
    job_url TEXT UNIQUE,
    job_description TEXT,
    date_found TEXT,
    source TEXT,
    ats_type TEXT,
    created_at TEXT DEFAULT (datetime('now'))
  );
  CREATE INDEX IF NOT EXISTS idx_jobs_score ON jobs(relevance_score DESC);
  CREATE INDEX IF NOT EXISTS idx_jobs_ghost ON jobs(is_ghost);
  "
  echo "  ✓ jobs.db schema created"
fi

# Profile embedding placeholder
if [ ! -f "$JOBS_DIR/db/profile_embedding.npy" ]; then
  python3 -c "
import numpy as np
vec = np.zeros(384, dtype=np.float32)  # all-MiniLM-L6-v2 dim
np.save('$JOBS_DIR/db/profile_embedding.npy', vec)
print('  ✓ profile_embedding.npy placeholder created (zeros — run /linkright-mem to generate real embedding)')
" 2>/dev/null || echo "  ⚠ numpy not available — profile_embedding.npy skipped"
fi

# pipeline.json
if [ ! -f "$JOBS_DIR/memory/pipeline.json" ]; then
  echo "[]" > "$JOBS_DIR/memory/pipeline.json"
  echo "  ✓ pipeline.json initialized (empty)"
else
  COUNT=$(python3 -c "import json; print(len(json.load(open('$JOBS_DIR/memory/pipeline.json'))))" 2>/dev/null || echo "?")
  echo "  ✓ pipeline.json exists ($COUNT opportunities)"
fi
```

## Step 9 — Health Check

Verify all 8 skills are wired. For each check: ✓ = ready, ✗ = broken (with fix), ⚠ = degraded (works but missing enhancement).

```bash
echo ""
echo "=== LinkRight Health Check ==="
echo ""

PASS=0; WARN=0; FAIL=0

hc() {
  local skill="$1" check="$2" fix="$3" level="${4:-fail}"
  if eval "$check" >/dev/null 2>&1; then
    echo "  ✓ $skill"
    ((PASS++)) || true
  else
    if [ "$level" = "warn" ]; then
      echo "  ⚠ $skill — $fix"
      ((WARN++)) || true
    else
      echo "  ✗ $skill — $fix"
      ((FAIL++)) || true
    fi
  fi
}

# linkright-mem
hc "linkright-mem: facts.md"         "[ -f ~/.linkright/memory/facts.md ]"                       "run /linkright-mem to onboard"
hc "linkright-mem: signals.md"       "[ -f ~/.linkright/memory/signals.md ]"                     "run /linkright-mem → ingest resume"
hc "linkright-mem: init_memory.py"   "[ -f ~/.claude/skills/linkright-mem/scripts/init_memory.py ]" "skill file missing"

# linkright-hunt
hc "linkright-hunt: jobs.db"         "[ -f ~/.linkright/jobs/db/jobs.db ]"                       "re-run step 5 above"
hc "linkright-hunt: pipeline.json"   "[ -f ~/.linkright/jobs/memory/pipeline.json ]"              "re-run step 5 above"
hc "linkright-hunt: scrape_jobs.py"  "[ -f ~/.linkright/jobs/scripts/scrape_jobs.py ] || [ -f ~/Downloads/*/scripts/scrape_jobs.py ]" "copy hunt scripts to ~/.linkright/jobs/scripts/" warn

# linkright-sync
hc "linkright-sync: facts.md"        "[ -f ~/.linkright/memory/facts.md ]"                       "run /linkright-mem first"
hc "linkright-sync: user_setup.md"   "[ -f ~/.linkright/user_setup.md ]"                         "re-run step 1"

# linkright-push
hc "linkright-push: git"             "command -v git"                                             "brew install git"
hc "linkright-push: gh"              "command -v gh"                                              "brew install gh"
hc "linkright-push: gh auth"         "gh auth status"                                             "run: gh auth login" warn

# linkright-interview
hc "linkright-interview: load_profile.sh" "[ -f ~/.claude/skills/linkright-interview-coach/scripts/load_profile.sh ]" "coach scripts missing"
hc "linkright-interview: kokoro_speak.sh" "[ -f ~/.claude/skills/linkright-interview-coach/scripts/kokoro_speak.sh ]" "TTS scripts missing"
hc "linkright-interview: signal_weights"  "[ -f ~/.claude/skills/linkright-interview-coach/state/sage-voice-config-schema.yaml ]" "coach state files missing" warn
hc "linkright-interview: round_catalogue" "[ -f ~/.claude/skills/linkright-interview-coach/lib/round_catalogue.md ]" "coach lib files missing"
hc "linkright-interview: TTS (say)"       "command -v say"                                        "no TTS available — text-only mode" warn

# linkright-companion
hc "linkright-companion: pipeline"   "[ -f ~/.linkright/jobs/memory/pipeline.json ]"              "run /linkright-hunt first" warn
hc "linkright-companion: morning_briefing.py" "[ -f ~/.claude/skills/linkright-companion/scripts/morning_briefing.py ]" "companion scripts missing"

# linkright-network
hc "linkright-network: user_setup"   "[ -f ~/.linkright/user_setup.md ]"                         "re-run step 1"
hc "linkright-network: voice_ref"    "[ -f ~/.claude/skills/linkright-network/references/ref_01_voice_profile.md ]" "network reference files missing"

# API keys (warn only)
source ~/.linkright/.env 2>/dev/null
[ -n "$GEMINI_API_KEY" ] && echo "  ✓ GEMINI_API_KEY set" || { echo "  ⚠ GEMINI_API_KEY — not set (A/V coaching degraded)"; ((WARN++)) || true; }

echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "  PASS: $PASS   WARN: $WARN   FAIL: $FAIL"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

if [ "$FAIL" -eq 0 ] && [ "$WARN" -eq 0 ]; then
  echo "  LinkRight fully operational ✓"
elif [ "$FAIL" -eq 0 ]; then
  echo "  LinkRight operational with warnings. Run /linkright-setup repair to fix."
else
  echo "  $FAIL critical issue(s) found. Run /linkright-setup repair to fix."
fi
```
